PREFIX ifc: <http://ifcowl.openbimstandards.org/IFC2X3_TC1#>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX express: <https://w3id.org/express#>

SELECT ?wall ?is_external
WHERE {
    ?wall a ifc:IfcWallStandardCase .
    ?rel ifc:relatedObjects_IfcRelDefines ?wall .
    ?rel ifc:relatingPropertyDefinition_IfcRelDefinesByProperties ?set .
    ?set ifc:hasProperties_IfcPropertySet ?prop .
    ?prop ifc:name_IfcProperty ?name_box .
    ?name_box express:hasString "IsExternal" .
    ?prop ifc:nominalValue_IfcPropertySingleValue ?value_box .
    ?value_box express:hasBoolean ?is_external .
}
